function proxy_on(){
    git config --global https.proxy http://127.0.0.1:1081

    export http_proxy="http://127.0.0.1:1081"
    export https_proxy="http://127.0.0.1:1081"

    git config --global --unset https.proxy
    git config --global --unset https.proxy

    unset http_proxy
    unset https_proxy
    unset all_proxy

    print_proxy
}
